create table student(roll_no int, name varchar(20), dept varchar(20));
insert into student values(1,'rutu','comp'),(2,'manasi','civil'),(3,'shruti','comp'),(4,'amey','civil'),(5,'shyam','comp'),(10,'shyam','comp');
select * from student;

create table stud_info(roll_no int,  name varchar(20), age int , marks numeric(10,2));
insert into stud_info values(1,'suarabh',19,80),(2,'shyam',19,95),(3,'renuka',18,85),(4,'manasi',19,89),(5,'suarabh',19,80);
select * from stud_info;

select * from student inner join stud_info on student.roll_no=stud_info.roll_no;

select student.roll_no,student.name,stud_info.name from student inner join stud_info on student.roll_no=stud_info.roll_no;

select * from student left join stud_info on student.roll_no=stud_info.roll_no;

select * from student right join stud_info on student.roll_no=stud_info.roll_no;

select * from student, stud_info where student.roll_no=stud_info.roll_no;

select * from stud_info where name in (select name from stud_info where marks>80);

create view studview as select roll_no,name,marks from stud_info where age=19;
select * from studview;
